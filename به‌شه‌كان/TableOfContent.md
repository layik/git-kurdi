# ناوه‌ڕۆك



٤ سه‌باره‌ی به‌م كتێبه‌

٥ **دامه‌زراندی گیت**

٥ دامه‌زراندن له‌ لینوكس

٦ دامه‌زراندن له‌ ماك

٧ ویندۆس

٨ مێژوویه‌كی كورتی گیت



١٠ **تێگه‌یشتن له‌ گیت**

١٠ گیت چیه‌؟

١١ فۆكس و دیزاین

١٣ جۆری ته‌نی گیت

٢٢ مۆدێلی داتای گیت

٢٨ برانچین و مێرجین

٤١ دار

٤٤ دایرێكته‌ری گیت

٤٨ ئیشكردن له‌ دایرێكته‌ری

٤٩ ئیندێكس

٥٤ **به‌كارهێنانی گیت**

٥٤ پێكهێنانی پرۆفایل

٥٥ بردن بۆ ریپۆی گیت

٥٧ نموونه‌ی ئیشكردنی نۆرماڵ

٦٣ لۆگ- مێژووی كۆمیت

٦٥ براوسین له‌ گیت

٧١ گه‌ڕان له‌ گیت

٧٢ گیت دیف (جیاوازی)

٧٥ برانچین

٧٧ مێرجینێكی ساده‌

٨٠ ڕیبه‌سه‌ین

٨٦ ستراشین 

٨٩ تاگین

٩١ هاویشتنه‌ ده‌ره‌وه‌ی گیت

٩٢ ئاگاداربوون له‌ گیت

٩٤ نموونه‌ی ئیشكردنی دیستریبیوتید 

١٠٥ شێری ریپۆكان

١٠٧ ریپۆ هۆستكراوه‌كان

١١١ **پوخته‌ی كۆماند**

١١١ گیتی سه‌ره‌تای

١١٥ پشكنینی ریپۆكان

١١٧ كه‌ره‌سته‌ی زیاتر

١١٩ **سه‌رچاوه‌ و ژێده‌ره‌كان**

١١٩ دۆكیۆمێنتی وێب

١٢٠ سكرین شۆت

١٢١ **پێداچوونه‌وه‌**
